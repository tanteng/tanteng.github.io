// CreateOnlineTable 创建统计用户在线时长表 func CreateOnlineTable() { year, month, _ := time.Now().Date() t := time.Date(year, month+1, 0, 0, 0, 0, 0, time.UTC) for day := 1; day <= t.Day(); day++ { date := time.Now().Format(“200601”) + fmt.Sprintf(“%02d”, day) tableName := fmt.Sprintf(utils.LoadIniConfig(“online”, “online”, “db”), date) template := utils.LoadIniConfig(“online”, “online_template”, “db”) sql := fmt.Sprintf(“CREATE TABLE IF NOT EXISTS %s LIKE %s”, tableName, template) fmt.Println(sql) utils.Query(sql, “sns_online_fish”) } }
